function range(start, end){
    array=[];
    if(start> end){
        counter=start;
        while(counter>= end){
            array.push(counter);
            counter-=1;
        }
    }else{
        counter=start;
        while(counter<= end){
            array.push(counter);
            counter+=1;
        }
    }
    return array;
}
